## Elevations

export const elevations = [
    {computed: "elevation-dropdown", className: "elevation-dropdown bg-[var(--colors-wafer-blue-wafer-blue-03)]"},
    {computed: "elevation-modal", className: "elevation-modal bg-[var(--colors-wafer-blue-wafer-blue-04)]"},
    {computed: "elevation-tooltip", className: "elevation-tooltip bg-[var(--colors-wafer-blue-wafer-blue-05)]"},
]

<div className="sb-unstyled">
    <div class="relative h-30">
      {elevations.map(({ computed, className, comment }, index) => {
          const postfix = computed.split("-").at(-1)
          return (
            <div key={index} style={{marginLeft: 30 * index, marginTop: 60 * index, zIndex: 100 * index}} className={`absolute flex flex-col py-sm px-lg gap-sm flex justify-start items-start w-50 h-20 rounded-sm ${className}`}>
                  <p className="font-mono font-bold">{computed} {comment && [":", <span className="opacity-dim-1">({comment})</span>]}</p>
                <p className="typo-caption-mono text-xs opacity-dim-1">shadow-{postfix} z-{postfix}</p>
            </div>
          )
      })}
    </div>
</div>
