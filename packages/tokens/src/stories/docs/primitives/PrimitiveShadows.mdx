# Primitive / Shadows (그림자)

export const spacing = [
    {varName: "--shadow-1"},
    {varName: "--shadow-2"},
    {varName: "--shadow-3"},
    {varName: "--shadow-4"},
    {varName: "--shadow-5"},
]

<div className="sb-unstyled pt-2xl flex flex-col gap-2xl items-center">
    {spacing.map(({ varName, comment }) => {
        const boxShadow = getCssVariable(varName)
        return (
          <div className="p-lg gap-lg flex justify-start items-center w-50 h-20 bg-brand rounded-sm" style={{boxShadow}}>
                <div className="flex flex-col gap-sm">
                    <p className="font-mono font-bold text-white">{varName} {comment && <span className="opacity-dim-1">({comment})</span>}</p>
                    <p className="typo-caption-mono text-white opacity-dim-1">
                        {boxShadow.split(",").reduce((acc, it) => [...acc, <p>{it},</p>], [])}
                    </p>
                </div>
            </div>
        )
    })}
</div>
